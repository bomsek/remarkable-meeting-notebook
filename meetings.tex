			\node[
			fill=white,
			rounded corners=10pt,
			minimum width=10cm,
			minimum height=4cm,
			align=center,
			yshift=3cm
			] at (current page.center)
			{
			{\fontsize{100}{120}\selectfont \textbf{Meetings}}

		\end{tikzpicture}
		
	\end{titlepage}
	
}
